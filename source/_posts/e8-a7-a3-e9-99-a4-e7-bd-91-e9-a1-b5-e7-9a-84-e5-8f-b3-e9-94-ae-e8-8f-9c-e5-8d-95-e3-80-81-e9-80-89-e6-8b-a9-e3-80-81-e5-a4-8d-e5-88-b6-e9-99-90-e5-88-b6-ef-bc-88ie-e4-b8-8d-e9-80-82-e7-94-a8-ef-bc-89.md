```javascript
function unlock(currWin){
    with(currWin.document){
        ondragstart = onselectstart = onselect = onbeforecopy = oncopy = onpaste = onkeydown = oncontextmenu = onmousedown = onmouseup = function(){return true};
        var all = getElementsByTagName(&#039;*&#039;);
        for ( var i = 0 ; i &lt; all.length ; i++){
            var win = all[i].contentWindow;//若win不为undefined，则all[i]为iframe
            if(win)
            try{
                win.document;//跨域会出错
                arguments.callee(win)
            }catch(e){
                console.log(&quot;跨域咯&quot;)
            }
            with(all[i]){
                ondragstart = onselectstart = onselect = onbeforecopy = oncopy = onpaste = onkeydown = oncontextmenu = onmousedown = onmouseup = function(){return true};all[i].style.cursor=&quot;auto&quot;
            }
        }
    }
}
unlock(this)
```
